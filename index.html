Телепорт игрока

Телепортирует игрока в указанное место при нажатии на часть.

local part = script.Parent

part.Touched:Connect(function(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid then
        local player = game.Players:GetPlayerFromCharacter(hit.Parent)
        if player then
            player.Character:SetPrimaryPartCFrame(CFrame.new(0, 10, 0))
        end
    end
end)

Случайный спавн

Спавнит игрока в случайном месте из заданных точек.

local spawnPoints = workspace.SpawnPoints:GetChildren()

game.Players.PlayerAdded:Connect(function(player)

Здоровье игрока

Устанавливает максимальное здоровье и восстанавливает его со временем.

game.Players.PlayerAdded:Connect(function(player)
    player.CharacterAdded:Connect(function(character)
        local humanoid = character:WaitForChild("Humanoid")
        humanoid.MaxHealth = 150
        humanoid.Health = humanoid.MaxHealth
        
        while humanoid and humanoid.Health > 0 do
            if humanoid.Health < humanoid.MaxHealth then
                humanoid.Health = humanoid.Health + 1
            end
            wait(1)
        end
    end)
end)

Система монет

Добавляет монеты игроку при касании и сохраняет их.

local part = script.Parent
local coinValue = 1

part.Touched:Connect(function(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid then
        local player = game.Players:GetPlayerFromCharacter(hit.Parent)
        if player then
            local coins = player:FindFirstChild("Coins") or Instance.new("IntValue")
            coins.Name = "Coins"
            coins.Value = coins.Value + coinValue
            coins.Parent = player
            part:Destroy()
        end
    end
end)

День/ночь

Цикл дня и ночи с изменением освещения.

local lighting = game:GetService("Lighting")
local cycleLength = 120 -- seconds

while true do
    for i = 0, cycleLength do
        local ratio = i / cycleLength
        lighting.ClockTime = 6 + (ratio * 12) % 24
        lighting.OutdoorAmbient = Color3.fromRGB(
            math.floor(100 + 155 * (1 - math.abs(ratio - 0.5) * 2),
            math.floor(100 + 155 * (1 - math.abs(ratio - 0.5) * 2),
            math.floor(100 + 155 * (1 - math.abs(ratio - 0.5) * 2)
        )
        wait(1)
    end
end

Граната

Создает гранату с таймером и взрывом.

local tool = script.Parent
local explosion = Instance.new("Explosion")
explosion.BlastPressure = 0
explosion.BlastRadius = 10

tool.Activated:Connect(function()
    local handle = tool:WaitForChild("Handle")
    local grenade = handle:Clone()
    grenade.CanCollide = true
    grenade.Anchored = false
    grenade.Parent = workspace
    
    local bodyVelocity = Instance.new("BodyVelocity")
    bodyVelocity.Velocity = tool.Parent:GetPrimaryPartCFrame().LookVector * 50
    bodyVelocity.Parent = grenade
    
    wait(3) -- Таймер гранаты
    
    explosion.Position = grenade.Position
    explosion.Parent = workspace
    grenade:Destroy()
end)

Система лидеров

Создает таблицу лидеров с очками убийств.

local leaderstats = Instance.new("Folder")
leaderstats.Name = "leaderstats"

local kills = Instance.new("IntValue")
kills.Name = "Kills"
kills.Value = 0
kills.Parent = leaderstats

game.Players.PlayerAdded:Connect(function(player)
    local ls = leaderstats:Clone()
    ls.Parent = player
    
    player.CharacterAdded:Connect(function(character)
        local humanoid = character:WaitForChild("Humanoid")
        humanoid.Died:Connect(function()
            local killer = humanoid:FindFirstChild("creator")
            if killer and killer.Value then
                killer.Value:FindFirstChild("Kills").Value += 1
            end
        end)
    end)
end)

Покупка за монеты

Позволяет игрокам покупать предметы за монеты.

local part = script.Parent
local itemToGive = game.ServerStorage:WaitForChild("SpecialSword")
local price = 100

part.Touched:Connect(function(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid then
        local player = game.Players:GetPlayerFromCharacter(hit.Parent)
        if player then
            local coins = player:FindFirstChild("Coins")
            if coins and coins.Value >= price then
                coins.Value = coins.Value - price
                local item = itemToGive:Clone()
                item.Parent = player.Backpack
            end
        end
    end
end)

Парящая платформа

Платформа, которая плавно движется вверх и вниз.

local platform = script.Parent
local startY = platform.Position.Y
local height = 5
local speed = 1

while true do
    local time = tick()
    local newY = startY + math.sin(time * speed) * height
    platform.Position = Vector3.new(platform.Position.X, newY, platform.Position.Z)
    wait(0.03)
end

Система дружеского огня

Включает/выключает дружеский огонь для команды.

local teams = game:GetService("Teams")
local friendlyFire = false

local function toggleFriendlyFire()
    friendlyFire = not friendlyFire
    for _, team in ipairs(teams:GetTeams()) do
        team.Neutral = friendlyFire
    end
    print("Friendly fire is now " .. (friendlyFire and "ON" or "OFF"))
end

game.Players.PlayerAdded:Connect(function(player)
    player.Chatted:Connect(function(message)
        if player:GetRankInGroup(1) > 0 and message:lower() == "/ff" then
            toggleFriendlyFire()
        end
    end)
end)

Система уровней

Добавляет игрокам уровни и опыт за действия.

game.Players.PlayerAdded:Connect(function(player)
    local leaderstats = Instance.new("Folder")
    leaderstats.Name = "leaderstats"
    leaderstats.Parent = player
    
    local level = Instance.new("IntValue")
    level.Name = "Level"
    level.Value = 1
    level.Parent = leaderstats
    
    local exp = Instance.new("IntValue")
    exp.Name = "Experience"
    exp.Value = 0
    exp.Parent = player
    
    local function checkLevelUp()
        local expNeeded = level.Value * 100
        if exp.Value >= expNeeded then
            level.Value = level.Value + 1
            exp.Value = exp.Value - expNeeded
        end
    end
    
    exp.Changed:Connect(checkLevelUp)
end)

Дождь

Создает эффект дождя в указанной области.

local rainArea = workspace.RainArea
local rainDrop = script.RainDrop

while true do
    local x = math.random(-50, 50) + rainArea.Position.X
    local z = math.random(-50, 50) + rainArea.Position.Z
    local drop = rainDrop:Clone()
    drop.Position = Vector3.new(x, rainArea.Position.Y + 50, z)
    drop.Parent = workspace
    drop.Anchored = false
    
    game:GetService("Debris"):AddItem(drop, 5)
    wait(0.05)
end

Система диалогов

Отображает диалоговое окно при касании NPC.

local npc = script.Parent
local dialogGui = script.DialogGui

npc.Touched:Connect(function(hit)
    local humanoid = hit.Parent:FindFirstChild("Humanoid")
    if humanoid then
        local player = game.Players:GetPlayerFromCharacter(hit.Parent)
        if player then
            local gui = dialogGui:Clone()
            gui.Parent = player.PlayerGui
            gui.Enabled = true
            
            script.DialogText.Text = "Hello, " .. player.Name .. "! Welcome to our game."
            
            gui.CloseButton.MouseButton1Click:Connect(function()
                gui:Destroy()
            end)
        end
    end
end)
